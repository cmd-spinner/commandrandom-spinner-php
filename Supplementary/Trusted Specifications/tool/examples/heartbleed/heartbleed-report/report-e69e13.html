564err:

668#ifndef OPENSSL_NO_BUF_FREELISTS
669/* On some platforms, malloc() performance is bad enough that you can't just
670 * free() and malloc() buffers all the time, so we need to use freelists from
671 * unused buffers. Currently, each freelist holds memory chunks of only a
672 * given size (list->chunklen); other sized chunks are freed and malloced.
673 * This doesn't help much if you're using many different SSL option settings
674 * with a given context. (The options affecting buffer size are
675 * max_send_fragment, read buffer vs write buffer,
676 * SSL_OP_MICROSOFT_BIG_WRITE_BUFFER, SSL_OP_NO_COMPRESSION, and
677 * SSL_OP_DONT_INSERT_EMPTY_FRAGMENTS.) Using a separate freelist for every
678 * possible size is not an option, since max_send_fragment can take on many
679 * different values.
680 *
681 * If you are on a platform with a slow malloc(), and you're using SSL
682 * connections with many different settings for these options, and you need to
683 * use the SSL_MOD_RELEASE_BUFFERS feature, you have a few options:
684 * - Link against a faster malloc implementation.
685 * - Use a separate SSL_CTX for each option set.
686 * - Improve this code.
687 */
688static void *
689freelist_extract(SSL_CTX *ctx, int for_read, int sz)
690 {
691 SSL3_BUF_FREELIST *list;
692 SSL3_BUF_FREELIST_ENTRY *ent = NULL((void*)0);
693 void *result = NULL((void*)0);
694
695 CRYPTO_w_lock(CRYPTO_LOCK_SSL_CTX)CRYPTO_lock(1|8,12,"s3_both.c",695);
696 list = for_read ? ctx->rbuf_freelist : ctx->wbuf_freelist;
697 if (list != NULL((void*)0) && sz == (int)list->chunklen)
698 ent = list->head;
699 if (ent != NULL((void*)0))
700 {
701 list->head = ent->next;
702 result = ent;
703 if (--list->len == 0)
704 list->chunklen = 0;
705 }
706 CRYPTO_w_unlock(CRYPTO_LOCK_SSL_CTX)CRYPTO_lock(2|8,12,"s3_both.c",706);
707 if (!result)
708 result = OPENSSL_malloc(sz)CRYPTO_malloc((int)sz,"s3_both.c",708);
709 return result;
710}
711
712static void
713freelist_insert(SSL_CTX *ctx, int for_read, size_t sz, void *mem)
714 {
715 SSL3_BUF_FREELIST *list;
716 SSL3_BUF_FREELIST_ENTRY *ent;
717
718 CRYPTO_w_lock(CRYPTO_LOCK_SSL_CTX)CRYPTO_lock(1|8,12,"s3_both.c",718);
719 list = for_read ? ctx->rbuf_freelist : ctx->wbuf_freelist;
720 if (list != NULL((void*)0) &&
721 (sz == list->chunklen || list->chunklen == 0) &&
722 list->len < ctx->freelist_max_len &&
723 sz >= sizeof(*ent))
724 {
725 list->chunklen = sz;
726 ent = mem;
727 ent->next = list->head;
728 list->head = ent;
729 ++list->len;
730 mem = NULL((void*)0);
731 }
732
733 CRYPTO_w_unlock(CRYPTO_LOCK_SSL_CTX)CRYPTO_lock(2|8,12,"s3_both.c",733);
734 if (mem)
735 OPENSSL_free(mem)CRYPTO_free(mem);
736 }
737#else
738#define freelist_extract(c,fr,sz) OPENSSL_malloc(sz)CRYPTO_malloc((int)sz,"s3_both.c",738)
739#define freelist_insert(c,fr,sz,m) OPENSSL_free(m)CRYPTO_free(m)
740#endif

742int ssl3_setup_read_buffer(SSL *s)
743 {
744 unsigned char *p;
745 size_t len,align=0,headerlen;
746
747 if (SSL_version(s) == DTLS1_VERSION0xFEFF || SSL_version(s) == DTLS1_BAD_VER0x0100)
748 headerlen = DTLS1_RT_HEADER_LENGTH13;
749 else
750 headerlen = SSL3_RT_HEADER_LENGTH5;
751
752#if defined(SSL3_ALIGN_PAYLOAD8) && SSL3_ALIGN_PAYLOAD8!=0
753 align = (-SSL3_RT_HEADER_LENGTH5)&(SSL3_ALIGN_PAYLOAD8-1);
754#endif
755
756 if (s->s3->rbuf.buf == NULL((void*)0))
757 {
758 len = SSL3_RT_MAX_PLAIN_LENGTH16384
759 + SSL3_RT_MAX_ENCRYPTED_OVERHEAD(256 + 64)
760 + headerlen + align;
761 if (s->options & SSL_OP_MICROSOFT_BIG_SSLV3_BUFFER0x00000020L)
762 {
763 s->s3->init_extra = 1;
764 len += SSL3_RT_MAX_EXTRA(16384);
765 }
766#ifndef OPENSSL_NO_COMP
767 if (!(s->options & SSL_OP_NO_COMPRESSION0x00020000L))
768 len += SSL3_RT_MAX_COMPRESSED_OVERHEAD1024;
769#endif
770 if ((p=freelist_extract(s->ctx, 1, len)) == NULL((void*)0))
771 goto err;
772 s->s3->rbuf.buf = p;
773 s->s3->rbuf.len = len;
774 }
775
776 s->packet= &(s->s3->rbuf.buf[0]);
777 return 1;
778
779err:
780 SSLerr(SSL_F_SSL3_SETUP_READ_BUFFER,ERR_R_MALLOC_FAILURE)ERR_put_error(20,(156),((1|64)),"s3_both.c",780);
781 return 0;
782 }

784int ssl3_setup_write_buffer(SSL *s)
785 {
786 unsigned char *p;
787 size_t len,align=0,headerlen;
788
789 if (SSL_version(s) == DTLS1_VERSION0xFEFF || SSL_version(s) == DTLS1_BAD_VER0x0100)
790 headerlen = DTLS1_RT_HEADER_LENGTH13 + 1;
791 else

793

796#endif
797
798 if (s->s3->wbuf.buf == NULL((void*)0))
799 {
800 len = s->max_send_fragment
801 + SSL3_RT_SEND_MAX_ENCRYPTED_OVERHEAD(16 + 64)
802 + headerlen + align;
803#ifndef OPENSSL_NO_COMP
804 if (!(s->options & SSL_OP_NO_COMPRESSION0x00020000L))
805 len += SSL3_RT_MAX_COMPRESSED_OVERHEAD1024;
806#endif
807 if (!(s->options & SSL_OP_DONT_INSERT_EMPTY_FRAGMENTS0x00000800L))
808 len += headerlen + align
809 + SSL3_RT_SEND_MAX_ENCRYPTED_OVERHEAD(16 + 64);
810
811 if ((p=freelist_extract(s->ctx, 0, len)) == NULL((void*)0))
812 goto err;
813 s->s3->wbuf.buf = p;
814 s->s3->wbuf.len = len;
815 }

819err:
820 SSLerr(SSL_F_SSL3_SETUP_WRITE_BUFFER,ERR_R_MALLOC_FAILURE)ERR_put_error(20,(291),((1|64)),"s3_both.c",820);
821 return 0;
822 }
