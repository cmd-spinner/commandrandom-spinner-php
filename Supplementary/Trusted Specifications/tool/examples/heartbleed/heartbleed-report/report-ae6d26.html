1549int BIO_dgram_sctp_wait_for_dry(BIO *b)
1550{
1551 int is_dry = 0;
1552 int n, sockflags, ret;
1553 union sctp_notification snp;
1554 struct msghdr msg;
1555 struct iovec iov;
1556#ifdef SCTP_EVENT
1557 struct sctp_event event;
1558#else
1559 struct sctp_event_subscribe event;
1560 socklen_t eventsize;
1561#endif
1562 bio_dgram_sctp_data *data = (bio_dgram_sctp_data *)b->ptr;
1563
1564 /* set sender dry event */
1565#ifdef SCTP_EVENT
1566 memset(&event, 0, sizeof(struct sctp_event))__builtin___memset_chk (&event, 0, sizeof(struct sctp_event
), __builtin_object_size (&event, 0));
1567 event.se_assoc_id = 0;
1568 event.se_type = SCTP_SENDER_DRY_EVENT;
1569 event.se_on = 1;
1570 ret = setsockopt(b->num, IPPROTO_SCTP132, SCTP_EVENT, &event, sizeof(struct sctp_event));
1571#else
1572 eventsize = sizeof(struct sctp_event_subscribe);
1573 ret = getsockopt(b->num, IPPROTO_SCTP132, SCTP_EVENTS, &event, &eventsize);
1574 if (ret < 0)
1575 return -1;
1576

1578
1579 ret = setsockopt(b->num, IPPROTO_SCTP132, SCTP_EVENTS, &event, sizeof(struct sctp_event_subscribe));
1580#endif

1582 return -1;

1584 /* peek for notification */
1585 memset(&snp, 0x00, sizeof(union sctp_notification))__builtin___memset_chk (&snp, 0x00, sizeof(union sctp_notification
), __builtin_object_size (&snp, 0));

1587 iov.iov_len = sizeof(union sctp_notification);
1588 msg.msg_name = NULL((void*)0);
1589 msg.msg_namelen = 0;
1590 msg.msg_iov = &iov;
1591 msg.msg_iovlen = 1;
1592 msg.msg_control = NULL((void*)0);
1593 msg.msg_controllen = 0;
1594 msg.msg_flags = 0;
1595
1596 n = recvmsg(b->num, &msg, MSG_PEEK0x2);
1597 if (n <= 0)
1598 {
1599 if ((n < 0) && (get_last_socket_error()(*__error()) != EAGAIN35) && (get_last_socket_error()(*__error()) != EWOULDBLOCK35))
1600 return -1;
1601 else
1602 return 0;
1603 }
1604
1605 /* if we find a notification, process it and try again if necessary */
1606 while (msg.msg_flags & MSG_NOTIFICATION)
1607 {
1608 memset(&snp, 0x00, sizeof(union sctp_notification))__builtin___memset_chk (&snp, 0x00, sizeof(union sctp_notification
), __builtin_object_size (&snp, 0));
1609 iov.iov_base = (char *)&snp;
1610 iov.iov_len = sizeof(union sctp_notification);
1611 msg.msg_name = NULL((void*)0);
1612 msg.msg_namelen = 0;
1613 msg.msg_iov = &iov;
1614 msg.msg_iovlen = 1;
1615 msg.msg_control = NULL((void*)0);
1616 msg.msg_controllen = 0;
1617 msg.msg_flags = 0;
1618
1619 n = recvmsg(b->num, &msg, 0);
1620 if (n <= 0)
1621 {
1622 if ((n < 0) && (get_last_socket_error()(*__error()) != EAGAIN35) && (get_last_socket_error()(*__error()) != EWOULDBLOCK35))
1623 return -1;
1624 else
1625 return is_dry;
1626 }
1627
1628 if (snp.sn_header.sn_type == SCTP_SENDER_DRY_EVENT)
1629 {
1630 is_dry = 1;
1631
1632 /* disable sender dry event */
1633#ifdef SCTP_EVENT
1634 memset(&event, 0, sizeof(struct sctp_event))__builtin___memset_chk (&event, 0, sizeof(struct sctp_event
), __builtin_object_size (&event, 0));
1635 event.se_assoc_id = 0;
1636 event.se_type = SCTP_SENDER_DRY_EVENT;
1637 event.se_on = 0;
1638 ret = setsockopt(b->num, IPPROTO_SCTP132, SCTP_EVENT, &event, sizeof(struct sctp_event));
1639#else
1640 eventsize = (socklen_t) sizeof(struct sctp_event_subscribe);
1641 ret = getsockopt(b->num, IPPROTO_SCTP132, SCTP_EVENTS, &event, &eventsize);
1642 if (ret < 0)
1643 return -1;
1644
1645 event.sctp_sender_dry_event = 0;
1646
1647 ret = setsockopt(b->num, IPPROTO_SCTP132, SCTP_EVENTS, &event, sizeof(struct sctp_event_subscribe));
1648#endif
1649 if (ret < 0)
1650 return -1;
1651 }
1652
1653#ifdef SCTP_AUTHENTICATION_EVENT
1654 if (snp.sn_header.sn_type == SCTP_AUTHENTICATION_EVENT)
1655 dgram_sctp_handle_auth_free_key_event(b, &snp);
1656#endif
1657
1658 if (data->handle_notifications != NULL((void*)0))
1659 data->handle_notifications(b, data->notification_context, (void*) &snp);
1660
1661 /* found notification, peek again */
1662 memset(&snp, 0x00, sizeof(union sctp_notification))__builtin___memset_chk (&snp, 0x00, sizeof(union sctp_notification
), __builtin_object_size (&snp, 0));
1663 iov.iov_base = (char *)&snp;
1664 iov.iov_len = sizeof(union sctp_notification);
1665 msg.msg_name = NULL((void*)0);
1666 msg.msg_namelen = 0;
1667 msg.msg_iov = &iov;
1668 msg.msg_iovlen = 1;
1669 msg.msg_control = NULL((void*)0);
1670 msg.msg_controllen = 0;
1671 msg.msg_flags = 0;
1672
1673 /* if we have seen the dry already, don't wait */
1674 if (is_dry)
1675 {
1676 sockflags = fcntl(b->num, F_GETFL, 0);
1677 fcntl(b->num, F_SETFL, O_NONBLOCK);
1678 }
1679
1680 n = recvmsg(b->num, &msg, MSG_PEEK0x2);
1681
1682 if (is_dry)
1683 {
1684 fcntl(b->num, F_SETFL, sockflags);
1685 }
1686
1687 if (n <= 0)
1688 {
1689 if ((n < 0) && (get_last_socket_error()(*__error()) != EAGAIN35) && (get_last_socket_error()(*__error()) != EWOULDBLOCK35))
1690 return -1;
1691 else
1692 return is_dry;
1693 }
1694 }
1695
1696 /* read anything else */
1697 return is_dry;
1698}
