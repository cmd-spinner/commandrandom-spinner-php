595 {
596 /* If ANY set type and pointer to value */
597 ASN1_TYPE *typ;
598 typ = (ASN1_TYPE *)*pval;
599 utype = typ->type;
600 *putype = utype;
601 pval = &typ->value.asn1_value;
602 }
603 else utype = *putype;
604
605 switch(utype)
606 {
607 case V_ASN1_OBJECT6:
608 otmp = (ASN1_OBJECT *)*pval;
609 cont = otmp->data;
610 len = otmp->length;
611 break;
612
613 case V_ASN1_NULL5:
614 cont = NULL((void*)0);
615 len = 0;
616 break;
617
618 case V_ASN1_BOOLEAN1:
619 tbool = (ASN1_BOOLEAN *)pval;
620 if (*tbool == -1)
621 return -1;
622 if (it->utype != V_ASN1_ANY-4)
623 {
624 /* Default handling if value == size field then omit */
625 if (*tbool && (it->size > 0))
626 return -1;
627 if (!*tbool && !it->size)
628 return -1;
629 }
630 c = (unsigned char)*tbool;
631 cont = &c;
632 len = 1;
633 break;
634
635 case V_ASN1_BIT_STRING3:
636 return i2c_ASN1_BIT_STRING((ASN1_BIT_STRING *)*pval,
637 cout ? &cout : NULL((void*)0));
638 break;
639
640 case V_ASN1_INTEGER2:
641 case V_ASN1_NEG_INTEGER(2 | 0x100):
642 case V_ASN1_ENUMERATED10:
643 case V_ASN1_NEG_ENUMERATED(10 | 0x100):
644 /* These are all have the same content format
645 * as ASN1_INTEGER
646 */
647 return i2c_ASN1_INTEGER((ASN1_INTEGER *)*pval,
648 cout ? &cout : NULL((void*)0));
649 break;
650
651 case V_ASN1_OCTET_STRING4:
652 case V_ASN1_NUMERICSTRING18:
653 case V_ASN1_PRINTABLESTRING19:
654 case V_ASN1_T61STRING20:
655 case V_ASN1_VIDEOTEXSTRING21:
656 case V_ASN1_IA5STRING22:
657 case V_ASN1_UTCTIME23:
658 case V_ASN1_GENERALIZEDTIME24:
659 case V_ASN1_GRAPHICSTRING25:
660 case V_ASN1_VISIBLESTRING26:
661 case V_ASN1_GENERALSTRING27:
662 case V_ASN1_UNIVERSALSTRING28:
663 case V_ASN1_BMPSTRING30:
664 case V_ASN1_UTF8STRING12:
665 case V_ASN1_SEQUENCE16:
666 case V_ASN1_SET17:
667 default:
668 /* All based on ASN1_STRING and handled the same */
669 strtmp = (ASN1_STRING *)*pval;
670 /* Special handling for NDEF */
671 if ((it->size == ASN1_TFLG_NDEF(0x1<<11))
672 && (strtmp->flags & ASN1_STRING_FLAG_NDEF0x010))
673 {
674 if (cout)
675 {
676 strtmp->data = cout;
677 strtmp->length = 0;
678 }
679 /* Special return code */
680 return -2;
681 }
682 cont = strtmp->data;
683 len = strtmp->length;
684
685 break;

(cout, 0));
