399 sarg.ndef_bio = NULL((void*)0);
400 sarg.boundary = NULL((void*)0);
401

403
404 if (aux->asn1_cb(ASN1_OP_DETACHED_PRE12, &val, it, &sarg) <= 0)

406

408 SMIME_crlf_copy(data, sarg.ndef_bio, flags);

417 {
418 tmpbio = BIO_pop(sarg.ndef_bio);
419 BIO_free(sarg.ndef_bio);
420 sarg.ndef_bio = tmpbio;
421 }
422
423 return rv;

649/* This is the big one: parse MIME header lines up to message body */
650
651#define MIME_INVALID0 0
652#define MIME_START1 1
653#define MIME_TYPE2 2
654#define MIME_NAME3 3
655#define MIME_VALUE4 4
656#define MIME_QUOTE5 5
657#define MIME_COMMENT6 6
658
659
660static STACK_OF(MIME_HEADER)struct stack_st_MIME_HEADER *mime_parse_hdr(BIO *bio)
661{
662 char *p, *q, c;
663 char *ntmp;
664 char linebuf[MAX_SMLEN1024];
665 MIME_HEADER *mhdr = NULL((void*)0);
666 STACK_OF(MIME_HEADER)struct stack_st_MIME_HEADER *headers;
667 int len, state, save_state = 0;
668
669 headers = sk_MIME_HEADER_new(mime_hdr_cmp)((struct stack_st_MIME_HEADER *)sk_new(((int (*)(const void *
, const void *)) ((1 ? (mime_hdr_cmp) : (int (*)(const MIME_HEADER
* const *, const MIME_HEADER * const *))0)))));
670 while ((len = BIO_gets(bio, linebuf, MAX_SMLEN1024)) > 0) {
671 /* If whitespace at line start then continuation line */
672 if(mhdr && isspace((unsigned char)linebuf[0])) state = MIME_NAME3;
673 else state = MIME_START1;
674 ntmp = NULL((void*)0);
675 /* Go through all characters */
676 for(p = linebuf, q = linebuf; (c = *p) && (c!='\r') && (c!='\n'); p++) {
677
678 /* State machine to handle MIME headers
679 * if this looks horrible that's because it *is*
680 */
681
682 switch(state) {
683 case MIME_START1:
684 if(c == ':') {
685 state = MIME_TYPE2;
686 *p = 0;
687 ntmp = strip_ends(q);
688 q = p + 1;
689 }
690 break;
691
692 case MIME_TYPE2:
693 if(c == ';') {
694 mime_debug("Found End Value\n");
695 *p = 0;
696 mhdr = mime_hdr_new(ntmp, strip_ends(q));
697 sk_MIME_HEADER_push(headers, mhdr)sk_push(((_STACK*) (1 ? (headers) : (struct stack_st_MIME_HEADER
*)0)), ((void*) (1 ? (mhdr) : (MIME_HEADER*)0)));
698 ntmp = NULL((void*)0);
699 q = p + 1;
700 state = MIME_NAME3;
701 } else if(c == '(') {
702 save_state = state;
703 state = MIME_COMMENT6;
704 }
705 break;
706
707 case MIME_COMMENT6:
708 if(c == ')') {
709 state = save_state;
710 }
711 break;
712
713 case MIME_NAME3:
714 if(c == '=') {
715 state = MIME_VALUE4;
716 *p = 0;
717 ntmp = strip_ends(q);
718 q = p + 1;
719 }
720 break ;
721
722 case MIME_VALUE4:
723 if(c == ';') {
724 state = MIME_NAME3;
725 *p = 0;
726 mime_hdr_addparam(mhdr, ntmp, strip_ends(q));
727 ntmp = NULL((void*)0);
728 q = p + 1;
729 } else if (c == '"') {
730 mime_debug("Found Quote\n");
731 state = MIME_QUOTE5;
732 } else if(c == '(') {
733 save_state = state;
734 state = MIME_COMMENT6;
735 }
736 break;
737
738 case MIME_QUOTE5:

741 state = MIME_VALUE4;
742 }
743 break;
744 }
745 }
746
747 if(state == MIME_TYPE2) {

749 sk_MIME_HEADER_push(headers, mhdr)sk_push(((_STACK*) (1 ? (headers) : (struct stack_st_MIME_HEADER
*)0)), ((void*) (1 ? (mhdr) : (MIME_HEADER*)0)));
750 } else if(state == MIME_VALUE4)

752 if(p == linebuf) break; /* Blank line means end of headers */
753}

755return headers;

764/* Strip a parameter of whitespace from start of param */
765static char *strip_start(char *name)
766{
767 char *p, c;
768 /* Look for first non white space or quote */
769 for(p = name; (c = *p) ;p++) {
770 if(c == '"') {
771 /* Next char is start of string if non null */
772 if(p[1]) return p + 1;
773 /* Else null string */
774 return NULL((void*)0);
775 }
776 if(!isspace((unsigned char)c)) return p;
777 }
778 return NULL((void*)0);
779}

783{
784 char *p, c;

786 /* Look for first non white space or quote */

788 c = *p;
789 if(c == '"') {
790 if(p - 1 == name) return NULL((void*)0);
791 *p = 0;
792 return name;
793 }
794 if(isspace((unsigned char)c)) *p = 0;

796 }
797 return NULL((void*)0);

916/* Check for a multipart boundary. Returns:
917 * 0 : no boundary
918 * 1 : part boundary
919 * 2 : final boundary
920 */
921static int mime_bound_check(char *line, int linelen, char *bound, int blen)
922{
923 if(linelen == -1) linelen = strlen(line);
924 if(blen == -1) blen = strlen(bound);
925 /* Quickly eliminate if line length too short */
926 if(blen + 2 > linelen) return 0;
927 /* Check for part boundary */
928 if(!strncmp(line, "--", 2) && !strncmp(line + 2, bound, blen)) {
929 if(!strncmp(line + blen + 2, "--", 2)) return 2;
930 else return 1;
931 }
932 return 0;
933}

935static int strip_eol(char *linebuf, int *plen)
936 {
937 int len = *plen;
938 char *p, c;
939 int is_eol = 0;
940 p = linebuf + len - 1;
941 for (p = linebuf + len - 1; len > 0; len--, p--)
942 {
943 c = *p;
944 if (c == '\n')
945 is_eol = 1;
946 else if (c != '\r')
947 break;
948 }
949 *plen = len;
950 return is_eol;
