74static unsigned int read_ledword(const unsigned char **in)
75 {
76 const unsigned char *p = *in;
77 unsigned int ret;
78 ret = *p++;
79 ret |= (*p++ << 8);
80 ret |= (*p++ << 16);
81 ret |= (*p++ << 24);
82 *in = p;
83 return ret;

139static int do_blob_header(const unsigned char **in, unsigned int length,
140 unsigned int *pmagic, unsigned int *pbitlen,
141 int *pisdss, int *pispub)
142 {
143 const unsigned char *p = *in;
144 if (length < 16)
145 return 0;
146 /* bType */
147 if (*p == MS_PUBLICKEYBLOB0x6)
148 {
149 if (*pispub == 0)
150 {
151 PEMerr(PEM_F_DO_BLOB_HEADER,ERR_put_error(9,(134),(119),"pvkfmt.c",152)
152 PEM_R_EXPECTING_PRIVATE_KEY_BLOB)ERR_put_error(9,(134),(119),"pvkfmt.c",152);
153 return 0;
154 }
155 *pispub = 1;
156 }
157 else if (*p == MS_PRIVATEKEYBLOB0x7)
158 {
159 if (*pispub == 1)
160 {
161 PEMerr(PEM_F_DO_BLOB_HEADER,ERR_put_error(9,(134),(120),"pvkfmt.c",162)
162 PEM_R_EXPECTING_PUBLIC_KEY_BLOB)ERR_put_error(9,(134),(120),"pvkfmt.c",162);
163 return 0;
164 }
165 *pispub = 0;
166 }
167 else
168 return 0;
169 p++;
170 /* Version */
171 if (*p++ != 0x2)
172 {
173 PEMerr(PEM_F_DO_BLOB_HEADER, PEM_R_BAD_VERSION_NUMBER)ERR_put_error(9,(134),(117),"pvkfmt.c",173);
174 return 0;
175 }
176 /* Ignore reserved, aiKeyAlg */
177 p+= 6;
178 *pmagic = read_ledword(&p);
179 *pbitlen = read_ledword(&p);
180 *pisdss = 0;
181 switch (*pmagic)
182 {
183
184 case MS_DSS1MAGIC0x31535344L:
185 *pisdss = 1;
186 case MS_RSA1MAGIC0x31415352L:
187 if (*pispub == 0)
188 {
189 PEMerr(PEM_F_DO_BLOB_HEADER,ERR_put_error(9,(134),(119),"pvkfmt.c",190)
190 PEM_R_EXPECTING_PRIVATE_KEY_BLOB)ERR_put_error(9,(134),(119),"pvkfmt.c",190);
191 return 0;
192 }
193 break;
194
195 case MS_DSS2MAGIC0x32535344L:
196 *pisdss = 1;
197 case MS_RSA2MAGIC0x32415352L:
198 if (*pispub == 1)
199 {
200 PEMerr(PEM_F_DO_BLOB_HEADER,ERR_put_error(9,(134),(120),"pvkfmt.c",201)
201 PEM_R_EXPECTING_PUBLIC_KEY_BLOB)ERR_put_error(9,(134),(120),"pvkfmt.c",201);
202 return 0;
203 }
204 break;
205
206 default:
207 PEMerr(PEM_F_DO_BLOB_HEADER, PEM_R_BAD_MAGIC_NUMBER)ERR_put_error(9,(134),(116),"pvkfmt.c",207);
208 return -1;
209 }
210 *in = p;

214static unsigned int blob_length(unsigned bitlen, int isdss, int ispub)
215 {
216 unsigned int nbyte, hnbyte;
217 nbyte = (bitlen + 7) >> 3;
218 hnbyte = (bitlen + 15) >> 4;
219 if (isdss)
220 {
221
222 /* Expected length: 20 for q + 3 components bitlen each + 24
223 * for seed structure.
224 */
225 if (ispub)
226 return 44 + 3 * nbyte;
227 /* Expected length: 20 for q, priv, 2 bitlen components + 24
228 * for seed structure.
229 */
230 else
231 return 64 + 2 * nbyte;
232 }
233 else
234 {
235 /* Expected length: 4 for 'e' + 'n' */
236 if (ispub)
237 return 4 + nbyte;
238 else
239 /* Expected length: 4 for 'e' and 7 other components.
240 * 2 components are bitlen size, 5 are bitlen/2
241 */
242 return 4 + 2*nbyte + 5*hnbyte;
243 }

312 unsigned int bitlen, int ispub)
313 {

315 EVP_PKEY *ret = NULL((void*)0);

317 BN_CTX *ctx = NULL((void*)0);

324 goto memerr;

326 goto memerr;
327 if (!read_lebn(&p, 20, &dsa->q))
328 goto memerr;

516 }

660 int skip_magic,
661 unsigned int *psaltlen, unsigned int *pkeylen)
662
663 {
664 const unsigned char *p = *in;
665 unsigned int pvk_magic, is_encrypted;
666 if (skip_magic)
667 {
668 if (length < 20)
669 {
670 PEMerr(PEM_F_DO_PVK_HEADER, PEM_R_PVK_TOO_SHORT)ERR_put_error(9,(136),(125),"pvkfmt.c",670);
671 return 0;
672 }
673 length -= 20;
Value stored to 'length' is never read
674 }
675 else
676 {
677 if (length < 24)
678 {
679 PEMerr(PEM_F_DO_PVK_HEADER, PEM_R_PVK_TOO_SHORT)ERR_put_error(9,(136),(125),"pvkfmt.c",679);
680 return 0;
681 }
682 length -= 24;
683 pvk_magic = read_ledword(&p);
684 if (pvk_magic != MS_PVKMAGIC0xb0b5f11eL)
685 {
686 PEMerr(PEM_F_DO_PVK_HEADER, PEM_R_BAD_MAGIC_NUMBER)ERR_put_error(9,(136),(116),"pvkfmt.c",686);
687 return 0;
688 }
689 }
690 /* Skip reserved */
691 p += 4;
692 /*keytype = */read_ledword(&p);
693 is_encrypted = read_ledword(&p);
694 *psaltlen = read_ledword(&p);
695 *pkeylen = read_ledword(&p);

699 PEMerr(PEM_F_DO_PVK_HEADER, PEM_R_INCONSISTENT_HEADER)ERR_put_error(9,(136),(121),"pvkfmt.c",699);
700 return 0;
701 }
702
703 *in = p;
704 return 1;
705 }

718 rv = 0;
719
720 EVP_MD_CTX_cleanup(&mctx);
721 return rv;
722 }

848 pem_password_cb *cb, void *u)

851 unsigned char *p, *salt = NULL((void*)0);
852 EVP_CIPHER_CTX cctx;
853 EVP_CIPHER_CTX_init(&cctx);
854 if (enclevel)
855 outlen += PVK_SALTLEN0x10;
856 pklen = do_i2b(NULL((void*)0), pk, 0);
857 if (pklen < 0)
858 return -1;
859 outlen += pklen;
860 if (!out)
861 return outlen;
862 if (*out)
863 p = *out;
864 else
865 {
866 p = OPENSSL_malloc(outlen)CRYPTO_malloc((int)outlen,"pvkfmt.c",866);
867 if (!p)
868 {
869 PEMerr(PEM_F_I2B_PVK,ERR_R_MALLOC_FAILURE)ERR_put_error(9,(137),((1|64)),"pvkfmt.c",869);
870 return -1;
871 }
872 *out = p;
873 }
874
875 write_ledword(&p, MS_PVKMAGIC0xb0b5f11eL);
876 write_ledword(&p, 0);
877 if (pk->type == EVP_PKEY_DSA116)
878 write_ledword(&p, MS_KEYTYPE_SIGN0x2);
879 else
880 write_ledword(&p, MS_KEYTYPE_KEYX0x1);
881 write_ledword(&p, enclevel ? 1 : 0);
882 write_ledword(&p, enclevel ? PVK_SALTLEN0x10: 0);
883 write_ledword(&p, pklen);
884 if (enclevel)

886 if (RAND_bytes(p, PVK_SALTLEN0x10) <= 0)
887 goto error;
888 salt = p;
889 p += PVK_SALTLEN0x10;
890 }
891 do_i2b(&p, pk, 0);
892 if (enclevel == 0)
893 return outlen;

895 {
896 char psbuf[PEM_BUFSIZE1024];
897 unsigned char keybuf[20];
898 int enctmplen, inlen;
899 if (cb)
900 inlen=cb(psbuf,PEM_BUFSIZE1024,1,u);
901 else
902 inlen=PEM_def_callback(psbuf,PEM_BUFSIZE1024,1,u);
903 if (inlen <= 0)
904 {
905 PEMerr(PEM_F_I2B_PVK,PEM_R_BAD_PASSWORD_READ)ERR_put_error(9,(137),(104),"pvkfmt.c",905);
906 goto error;
907 }
908 if (!derive_pvk_key(keybuf, salt, PVK_SALTLEN0x10,

912 memset(keybuf + 5, 0, 11)__builtin___memset_chk (keybuf + 5, 0, 11, __builtin_object_size
(keybuf + 5, 0));
913 p = salt + PVK_SALTLEN0x10 + 8;

915 goto error;
916 OPENSSL_cleanse(keybuf, 20);
917 if (!EVP_DecryptUpdate(&cctx, p, &enctmplen, p, pklen - 8))
918 goto error;
919 if (!EVP_DecryptFinal_ex(&cctx, p + enctmplen, &enctmplen))
920 goto error;
921 }
922 EVP_CIPHER_CTX_cleanup(&cctx);
923 return outlen;
924
925 error:

928 }
