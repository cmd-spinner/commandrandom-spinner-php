613 ASN1_GEN_STR("FORMAT", ASN1_GEN_FLAG_FORMAT){"FORMAT", sizeof("FORMAT") - 1, (0x10000|8)},
614 };
615
616 if (len == -1)
617 len = strlen(tagstr);
618
619 tntmp = tnst;
620 for (i = 0; i < sizeof(tnst) / sizeof(struct tag_name_st); i++, tntmp++)
621 {
622 if ((len == tntmp->len) && !strncmp(tntmp->strnam, tagstr, len))
623 return tntmp->tag;
624 }
625
626 return -1;

629static ASN1_TYPE *asn1_str2type(const char *str, int format, int utype)
630 {
631 ASN1_TYPE *atmp = NULL((void*)0);
632
633 CONF_VALUE vtmp;
634
635 unsigned char *rdata;
636 long rdlen;
637
638 int no_unused = 1;
639
640 if (!(atmp = ASN1_TYPE_new()))
641 {
642 ASN1err(ASN1_F_ASN1_STR2TYPE, ERR_R_MALLOC_FAILURE)ERR_put_error(13,(179),((1|64)),"asn1_gen.c",642);
643 return NULL((void*)0);
644 }
645
646 if (!str)
647 str = "";
648
649 switch(utype)
650 {
651
652 case V_ASN1_NULL5:
653 if (str && *str)
654 {
655 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_NULL_VALUE)ERR_put_error(13,(179),(182),"asn1_gen.c",655);
656 goto bad_form;
657 }
658 break;
659
660 case V_ASN1_BOOLEAN1:
661 if (format != ASN1_GEN_FORMAT_ASCII1)
662 {
663 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_NOT_ASCII_FORMAT)ERR_put_error(13,(179),(190),"asn1_gen.c",663);
664 goto bad_form;
665 }
666 vtmp.name = NULL((void*)0);
667 vtmp.section = NULL((void*)0);
668 vtmp.value = (char *)str;
669 if (!X509V3_get_value_bool(&vtmp, &atmp->value.boolean))
670 {
671 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_BOOLEAN)ERR_put_error(13,(179),(176),"asn1_gen.c",671);
672 goto bad_str;
673 }
674 break;
675
676 case V_ASN1_INTEGER2:
677 case V_ASN1_ENUMERATED10:
678 if (format != ASN1_GEN_FORMAT_ASCII1)
679 {
680 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_INTEGER_NOT_ASCII_FORMAT)ERR_put_error(13,(179),(185),"asn1_gen.c",680);
681 goto bad_form;
682 }
683 if (!(atmp->value.integer = s2i_ASN1_INTEGER(NULL((void*)0), (char *)str)))
684 {
685 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_INTEGER)ERR_put_error(13,(179),(180),"asn1_gen.c",685);
686 goto bad_str;
687 }
688 break;
689
690 case V_ASN1_OBJECT6:
691 if (format != ASN1_GEN_FORMAT_ASCII1)
692 {
693 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_OBJECT_NOT_ASCII_FORMAT)ERR_put_error(13,(179),(191),"asn1_gen.c",693);
694 goto bad_form;
695 }
696 if (!(atmp->value.object = OBJ_txt2obj(str, 0)))
697 {
698 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_OBJECT)ERR_put_error(13,(179),(183),"asn1_gen.c",698);
699 goto bad_str;
700 }
701 break;
702
703 case V_ASN1_UTCTIME23:
704 case V_ASN1_GENERALIZEDTIME24:
705 if (format != ASN1_GEN_FORMAT_ASCII1)
706 {
707 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_TIME_NOT_ASCII_FORMAT)ERR_put_error(13,(179),(193),"asn1_gen.c",707);
708 goto bad_form;
709 }
710 if (!(atmp->value.asn1_string = ASN1_STRING_new()))
711 {
712 ASN1err(ASN1_F_ASN1_STR2TYPE, ERR_R_MALLOC_FAILURE)ERR_put_error(13,(179),((1|64)),"asn1_gen.c",712);
713 goto bad_str;
714 }
715 if (!ASN1_STRING_set(atmp->value.asn1_string, str, -1))
716 {
717 ASN1err(ASN1_F_ASN1_STR2TYPE, ERR_R_MALLOC_FAILURE)ERR_put_error(13,(179),((1|64)),"asn1_gen.c",717);
718 goto bad_str;
719 }
720 atmp->value.asn1_string->type = utype;
721 if (!ASN1_TIME_check(atmp->value.asn1_string))
722 {
723 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_TIME_VALUE)ERR_put_error(13,(179),(184),"asn1_gen.c",723);
724 goto bad_str;
725 }
726
727 break;
728
729 case V_ASN1_BMPSTRING30:
730 case V_ASN1_PRINTABLESTRING19:
731 case V_ASN1_IA5STRING22:
732 case V_ASN1_T61STRING20:
733 case V_ASN1_UTF8STRING12:
734 case V_ASN1_VISIBLESTRING26:
735 case V_ASN1_UNIVERSALSTRING28:
736 case V_ASN1_GENERALSTRING27:
737 case V_ASN1_NUMERICSTRING18:
738
739 if (format == ASN1_GEN_FORMAT_ASCII1)
740 format = MBSTRING_ASC(0x1000|1);
741 else if (format == ASN1_GEN_FORMAT_UTF82)
742 format = MBSTRING_UTF8(0x1000);
743 else
744 {
745 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_FORMAT)ERR_put_error(13,(179),(177),"asn1_gen.c",745);
746 goto bad_form;
747 }
748
749
750 if (ASN1_mbstring_copy(&atmp->value.asn1_string, (unsigned char *)str,
751 -1, format, ASN1_tag2bit(utype)) <= 0)
752 {
753 ASN1err(ASN1_F_ASN1_STR2TYPE, ERR_R_MALLOC_FAILURE)ERR_put_error(13,(179),((1|64)),"asn1_gen.c",753);
754 goto bad_str;
755 }
756
757
758 break;
759
760 case V_ASN1_BIT_STRING3:
761
762 case V_ASN1_OCTET_STRING4:
763
764 if (!(atmp->value.asn1_string = ASN1_STRING_new()))
765 {
766 ASN1err(ASN1_F_ASN1_STR2TYPE, ERR_R_MALLOC_FAILURE)ERR_put_error(13,(179),((1|64)),"asn1_gen.c",766);
767 goto bad_form;
768 }
769
770 if (format == ASN1_GEN_FORMAT_HEX3)
771 {
772
773 if (!(rdata = string_to_hex((char *)str, &rdlen)))
774 {
775 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_HEX)ERR_put_error(13,(179),(178),"asn1_gen.c",775);
776 goto bad_str;
777 }
778
779 atmp->value.asn1_string->data = rdata;
780 atmp->value.asn1_string->length = rdlen;
781 atmp->value.asn1_string->type = utype;
782
783 }
784 else if (format == ASN1_GEN_FORMAT_ASCII1)
785 ASN1_STRING_set(atmp->value.asn1_string, str, -1);
786 else if ((format == ASN1_GEN_FORMAT_BITLIST4) && (utype == V_ASN1_BIT_STRING3))
787 {
788 if (!CONF_parse_list(str, ',', 1, bitstr_cb, atmp->value.bit_string))
789 {
790 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_LIST_ERROR)ERR_put_error(13,(179),(188),"asn1_gen.c",790);
791 goto bad_str;
792 }
793 no_unused = 0;
794
795 }
796 else
797 {
798 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_ILLEGAL_BITSTRING_FORMAT)ERR_put_error(13,(179),(175),"asn1_gen.c",798);
799 goto bad_form;
800 }
801
802 if ((utype == V_ASN1_BIT_STRING3) && no_unused)
803 {
804 atmp->value.asn1_string->flags
805 &= ~(ASN1_STRING_FLAG_BITS_LEFT0x08|0x07);
806 atmp->value.asn1_string->flags
807 |= ASN1_STRING_FLAG_BITS_LEFT0x08;
808 }
809
810
811 break;
812
813 default:
814 ASN1err(ASN1_F_ASN1_STR2TYPE, ASN1_R_UNSUPPORTED_TYPE)ERR_put_error(13,(179),(196),"asn1_gen.c",814);
815 goto bad_str;
816 break;
817 }
818
819
820 atmp->type = utype;
821 return atmp;
822
823
824 bad_str:
825 ERR_add_error_data(2, "string=", str);
826 bad_form:
827
828 ASN1_TYPE_free(atmp);
829 return NULL((void*)0);
830
831 }

833static int bitstr_cb(const char *elem, int len, void *bitstr)
834 {
835 long bitnum;
836 char *eptr;
837 if (!elem)
838 return 0;
839 bitnum = strtoul(elem, &eptr, 10);
840 if (eptr && *eptr && (eptr != elem + len))
841 return 0;
842 if (bitnum < 0)
843 {
844 ASN1err(ASN1_F_BITSTR_CB, ASN1_R_INVALID_NUMBER)ERR_put_error(13,(180),(187),"asn1_gen.c",844);
845 return 0;
846 }
847 if (!ASN1_BIT_STRING_set_bit(bitstr, bitnum, 1))
848 {
849 ASN1err(ASN1_F_BITSTR_CB, ERR_R_MALLOC_FAILURE)ERR_put_error(13,(180),((1|64)),"asn1_gen.c",849);
850 return 0;
851 }
852 return 1;
853 }
