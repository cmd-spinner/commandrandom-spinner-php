146static void sha1_update(SHA_CTX *c,const void *data,size_t len)
147{ const unsigned char *ptr = data;
148 size_t res;
149
150 if ((res = c->num)) {
151 res = SHA_CBLOCK(16*4)-res;
152 if (len<res) res=len;
153 SHA1_Updatesha1_update (c,ptr,res);
154 ptr += res;
155 len -= res;
156 }

158 res = len % SHA_CBLOCK(16*4);
159 len -= res;
160
161 if (len) {
162 sha1_block_data_order(c,ptr,len/SHA_CBLOCK(16*4));
163
164 ptr += len;
165 c->Nh += len>>29;
166 c->Nl += len<<=3;
167 if (c->Nl<(unsigned int)len) c->Nh++;
168 }
169
170 if (res)
171 SHA1_Updatesha1_update(c,ptr,res);
